            <h2 class="text-2xl font-semibold text-emerald-800 mb-4">The Bioscience Problem</h2>
            <p class="mb-4 leading-relaxed">
                In Malaysian oil palm agriculture, tracking the mass accumulation of **Fresh Fruit Bunches (FFB)** is vital for predicting yield and scheduling harvests. FFB growth follows a sigmoidal (S-shaped) curve. Early on, cellular division is slow; mid-season sees a massive surge in oil synthesis and weight gain; finally, growth levels off as the fruit matures.
            </p>
            <p class="mb-4 leading-relaxed">
                By applying the <strong>first derivative $f'(x)$</strong> to the growth function $f(x)$, we calculate the <em>instantaneous growth rate</em> (kg/day). Identifying the peak of the derivative tells agrotechnologists exactly when the fruit is accumulating mass the fastest, mapping directly to peak nutritional and oil-filling stages.
            </p>
            
            <div class="bg-emerald-50 border-l-4 border-emerald-600 p-4 rounded-r-lg my-4">

                <div class="grid grid-cols-1 md:grid-cols-2 gap-4 text-sm md:text-base">

                        <span class="block text-xs font-semibold text-emerald-700 uppercase">Growth Function (Logistic)</span>

    <script>
        // --- CONSTANTS & BIOLOGICAL PARAMETERS ---
        const L = 25;      // Carrying capacity (Max weight of bunch in kg)
        const k = 0.06;    // Growth rate coefficient 
        const x0 = 75;     // Inflection point (Day of fastest growth, mid-point of 150 days)

        // --- MATH FUNCTIONS ---
        // Logistic growth function: f(x)
        function f(x) {
            return L / (1 + Math.exp(-k * (x - x0)));
        }

        // Analytical derivative: f'(x)
        function fPrime(x) {
            const fx = f(x);
            return k * fx * (1 - (fx / L));
        }

        // --- DATA GENERATION ---
        const days = [];
        const weightData = [];
        const rateData = [];

        for (let day = 0; day <= 150; day += 5) {
            days.push(day);
            weightData.push(f(day).toFixed(2));
            rateData.push(fPrime(day).toFixed(3));
        }

        // --- CHARTJS CONFIGURATION ---
        const ctx = document.getElementById('growthChart').getContext('2d');
        
        const growthChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: days,
                datasets: [
                    {
                        label: 'Bunch Weight (f(x) in kg)',
                        data: weightData,
                        borderColor: '#059669',
                        backgroundColor: 'rgba(5, 150, 105, 0.1)',
                        yAxisID: 'yWeight',
                        tension: 0.3,
                        pointRadius: 4,
                        pointHoverRadius: 7
                    },
                    {
                        label: 'Growth Rate (f\'(x) in kg/day)',
                        data: rateData,
                        borderColor: '#2563eb',
                        backgroundColor: 'transparent',
                        borderDash: [5, 5],
                        yAxisID: 'yRate',
                        tension: 0.3,
                        pointRadius: 4,
                        pointHoverRadius: 7
                    }
                ]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                interaction: {
                    mode: 'index',
                    intersect: false,
                },
                plugins: {
                    legend: {
                        position: 'top',
                    }
                },
                scales: {
                    x: {
                        title: {
                            display: true,
                            text: 'Days After Flowering (Anthesis)',
                            font: { weight: 'bold' }
                        }
                    },
                    yWeight: {
                        type: 'linear',
                        display: true,
                        position: 'left',
                        title: {
                            display: true,
                            text: 'Total Weight (kg)',
                            font: { weight: 'bold', color: '#059669' }
                        },
                        min: 0,
                        max: 28
                    },
                    yRate: {
                        type: 'linear',
                        display: true,
                        position: 'right',
                        title: {
                            display: true,
                            text: 'Rate of Change (kg/day)',
                            font: { weight: 'bold', color: '#2563eb' }
                        },
                        min: 0,
                        max: 0.5,
                        grid: {
                            drawOnChartArea: false, // only want the grid lines for one axis
                        },
                    },
                },
                // Triggered when user clicks anywhere on the canvas
                onClick: (e) => {
                    const points = growthChart.getElementsAtEventForMode(e, 'index', { intersect: false }, true);
                    
                    if (points.length > 0) {
                        const index = points[0].index;
                        const targetDay = days[index];
                        
                        updateAnalysisPanel(targetDay);
                    }
                }
            }
        });

        // --- INTERACTIVE PANEL UPDATE LOGIC ---
        function updateAnalysisPanel(day) {
            const currentWeight = f(day);
            const currentRate = fPrime(day);

            // Toggle UI elements
            document.getElementById('instructionPrompt').classList.add('hidden');
            document.getElementById('dataMetrics').classList.remove('hidden');
            document.getElementById('laymanExplanation').classList.remove('hidden');

            // Inject core data strings
            document.getElementById('metricDay').innerText = `Day ${day}`;
            document.getElementById('metricWeight').innerText = `${currentWeight.toFixed(2)} kg`;
            document.getElementById('metricRate').innerText = `${currentRate.toFixed(3)} kg per day`;

            // Dynamic Bio-layman generation logic based on timeline location
            let explanation = "";
            if (day < 50) {
                explanation = `The bunch is in its lag phase. Cells are dividing internally, but weight gain is minimal (${currentRate.toFixed(3)} kg/day). Harvesting or major feeding right now yields little value.`;
            } else if (day >= 50 && day <= 100) {
                if (day >= 70 && day <= 80) {
                    explanation = `🔥 **Peak Efficiency!** At Day ${day}, the fruit is at its curve inflection point. The derivative is maximized. The bunch is putting on a massive ${currentRate.toFixed(3)} kg every single day due to active oil synthesis.`;
                } else {
                    explanation = `The plant is in its rapid exponential growth window. The derivative is very high, meaning fertilizer interventions here intensely push bunch sizing.`;
                }
            } else {
                explanation = `The bunch growth is hitting its asymptote ($L=25\\text{kg}$). The derivative is dropping close to zero (${currentRate.toFixed(3)} kg/day), meaning growth has stopped. The fruit is ripe, sugars are locked, and it is ready for the harvesting hook.`;
            }

            document.getElementById('explanationText').innerHTML = explanation;
        }
    </script>
